\section{Motivação e problema}

\begin{frame}{Motivação}
  \begin{itemize}
    \item Sensores LPFG exigem, em geral, interrogação espectral cara.
    \item Alternativa: array de FBGs + aprendizado de máquina para estimar $\lambda_{res}$.
    \item No trabalho do Barino, a atenção sugere que \textbf{nem todas as FBGs importam igualmente}.
  \end{itemize}
  \vspace{0.6em}
  \textbf{Pergunta de reconhecimento de padrões:}\\
  a partir das potências das 13 FBGs, \emph{quais} são as relevantes para localizar o vale?
\end{frame}

\begin{frame}{Objetivo deste trabalho}
  \begin{block}{Parte 1 --- Classificação (foco)}
    Predizer uma \textbf{máscara multi-rótulo} $y\in\{0,1\}^{13}$:\\
    os $k$ FBGs mais próximos de $\lambda_{res}$ ($k{=}4$).
  \end{block}
  \begin{block}{Parte 2 --- Interrogação (contexto)}
    Estimar $\lambda_{res}$ (já feita no paper do Barino; não refeita aqui).
  \end{block}
  \vspace{0.4em}
  Entregas da disciplina: artigo IEEEtran + apresentação (15~min).
\end{frame}

\begin{frame}{Formulação do problema de RP}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Entrada} $\mathbf{x}\in\mathbb{R}^{13}$\\
      potências ópticas normalizadas.
      \vspace{0.5em}

      \textbf{Rótulo} $y\in\{0,1\}^{13}$\\[0.3em]
      $e_i = |\lambda_{\mathrm{FBG},i}-\lambda_{res}|$\\[0.2em]
      $y_i=1$ nos $k$ menores $e_i$.
      \vspace{0.5em}

      \textbf{Predição:} top-$k$ nos escores\\
      (cardinalidade fixa $=4$).
    \end{column}
    \begin{column}{0.46\textwidth}
      \includegraphics[width=\linewidth]{fig_schema_lpg_fbg.png}
    \end{column}
  \end{columns}
\end{frame}

\section{Dados e pré-processamento}

\begin{frame}{Origem dos dados}
  \begin{itemize}
    \item Fonte: \texttt{measured.dataset} (dados \textbf{reais}, autorizados).
    \item Contexto: demodulação LPFG com array FBG (Barino et al.).
    \item Cada amostra corresponde a uma configuração medida do sistema óptico.
  \end{itemize}
  \vspace{0.8em}
  \begin{table}
    \centering
    \begin{tabular}{lcc}
      \toprule
      Campo & Dimensão & Uso em RP \\
      \midrule
      \texttt{input\_strength} & 13 & atributos ($X$) \\
      \texttt{wl\_bragg} & 13 & gera rótulo \\
      \texttt{target} ($\lambda_{res}$) & 1 & gera rótulo \\
      \bottomrule
    \end{tabular}
  \end{table}
\end{frame}

\begin{frame}{Pré-processamento}
  \begin{enumerate}
    \item \textbf{Filtro de faixa:} manter $1515 < \lambda_{res} < 1585$~nm\\
          $8200 \rightarrow 7300$ amostras (900 removidas).
    \item \textbf{Normalização} (por amostra, critério do Barino):\\
          $X \leftarrow X-\min(X)$; \quad $X \leftarrow X/\sum X$.
    \item \textbf{Rótulo top-4} a partir de $|\lambda_{\mathrm{FBG}}-\lambda_{res}|$.
  \end{enumerate}
  \vspace{0.6em}
  \begin{alertblock}{Observação}
    A máscara é \textbf{desbalanceada} por canal:\\
    FBG~4 $\approx 68\%$ positivos; FBG~12 $\approx 1{,}4\%$.
  \end{alertblock}
\end{frame}

\begin{frame}{Atributos e rótulos (visão de RP)}
  \begin{itemize}
    \item \textbf{Atributos:} 13 potências normalizadas (protocolo principal).
    \item \textbf{Rótulo:} vetor binário 13D (multi-rótulo), exatamente 4 uns.
    \item \textbf{Ablação:} também testamos concatenar $\lambda_{\mathrm{FBG}}$ à entrada.
    \item Há 10 padrões de máscara distintos no conjunto filtrado.
  \end{itemize}
  \vspace{0.5em}
  Coerência espacial do rótulo: correlação posição FBG $\times$ $\lambda_{res}$\\
  quando o canal é positivo $\approx 0{,}992$.
\end{frame}

\section{Metodologia experimental}

\begin{frame}{Protocolo de avaliação}
  \begin{itemize}
    \item Semente fixa: \texttt{random\_state=42}.
    \item Hold-out estratificado (20\%): \textbf{1460} amostras --- só relatório final.
    \item Desenvolvimento: \textbf{5840} amostras.
    \item CV: \texttt{StratifiedKFold} (5 folds) por quantis de $\lambda_{res}$.
    \item Em cada fold da CV: treino 4672 / teste 1168.
    \item Nested CV para hiperparâmetros (scorer $=$ Jaccard top-$k$).
    \item Sem SMOTE; hold-out nunca usado no ajuste.
  \end{itemize}
\end{frame}

\begin{frame}{Classificadores}
  Seis métodos da disciplina, adaptados a multi-rótulo:
  \vspace{0.4em}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{itemize}
        \item kNN (+ scaler)
        \item SVM RBF one-vs-rest
        \item MLP (+ scaler)
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{itemize}
        \item Random Forest
        \item AdaBoost one-vs-rest
        \item MQ / Ridge + top-$k$
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{0.8em}
  \textbf{Métricas:} Jaccard (principal), exact match, F1 micro/macro, Hamming.\\
  Predição alinhada ao problema: \textbf{top-4 por score}.
\end{frame}

\begin{frame}{Ajuste de hiperparâmetros (resumo)}
  \begin{itemize}
    \item Nested CV no desenvolvimento (busca só no treino de cada fold).
    \item Consenso estável do SVM: $C{=}10$, $\gamma{=}0{,}1$ (todos os folds).
    \item Maior ganho do tuning: SVM Jaccard $0{,}961\rightarrow 0{,}974$ (+0{,}013).
    \item Demais métodos: variação pequena após o ajuste.
  \end{itemize}
\end{frame}

\section{Conclusão}

\begin{frame}{Conclusão}
  \begin{itemize}
    \item Problema de RP bem posto: multi-rótulo top-4 sobre 13 potências.
    \item Dados reais (7300 amostras úteis), CV + hold-out isolado.
    \item Melhor método: \textbf{SVM RBF one-vs-rest} (Jaccard $\approx 0{,}975$ no hold-out).
    \item MQ serve como baseline linear mais fraco.
    \item Próximo passo natural: acoplar a máscara à regressão de $\lambda_{res}$.
  \end{itemize}
\end{frame}
