\section*{Quest\~ao 1 --- Fun\c{c}\~ao discriminante de Bayes}

\paragraph{Enunciado.} Obter a fun\c{c}\~ao discriminante baseada na
regra de Bayes para um problema de classifica\c{c}\~ao com duas classes
equiprov\'aveis dadas por
\[
C_1:\;\mathcal{N}(\mu_1,\Sigma),\qquad
C_2:\;\mathcal{N}(\mu_2,\Sigma),\qquad
P(C_1)=P(C_2)=\tfrac12.
\]

\paragraph{Solu\c{c}\~ao.} A regra de Bayes escolhe a classe que
maximiza a probabilidade a posteriori. Como $\ln$ \'e crescente,
podemos comparar
\[
g_i(\mathbf{x})=\ln p(\mathbf{x}\mid C_i)+\ln P(C_i).
\]
Para Gaussianas em dimens\~ao $d$,
\[
g_i(\mathbf{x})=
-\tfrac12(\mathbf{x}-\mu_i)^\top\Sigma_i^{-1}(\mathbf{x}-\mu_i)
-\tfrac{d}{2}\ln(2\pi)-\tfrac12\ln|\Sigma_i|+\ln P(C_i).
\]
Como $P(C_1)=P(C_2)$ e $\Sigma_1=\Sigma_2=I$, todos os termos
constantes em $i$ se cancelam na compara\c{c}\~ao. Resta
$-\tfrac12\|\mathbf{x}-\mu_i\|^2$; expandindo o quadrado, o termo
$-\tfrac12\mathbf{x}^\top\mathbf{x}$ tamb\'em \'e comum, de modo que basta comparar
\[
g_i(\mathbf{x})=\mu_i^\top\mathbf{x}-\tfrac12\mu_i^\top\mu_i.
\]
Substituindo $\mu_1=\mathbf{0}$ e $\mu_2=(3,3)^\top$,
$g_1(\mathbf{x})=0$ e $g_2(\mathbf{x})=3x_1+3x_2-9$. A fronteira ocorre quando
$g_1=g_2$, isto \'e, $3x_1+3x_2-9=0\Leftrightarrow x_1+x_2=3$. A
fun\c{c}\~ao discriminante bin\'aria \'e
\[
h(\mathbf{x})=3x_1+3x_2-9,\qquad
\begin{cases}
h(\mathbf{x})<0\Rightarrow \mathbf{x}\in C_1,\\
h(\mathbf{x})\ge 0\Rightarrow \mathbf{x}\in C_2.
\end{cases}
\]

\paragraph{Interpreta\c{c}\~ao geom\'etrica.} A reta $x_1+x_2=3$ tem
normal $(1,1)^\top$, paralela a $\mu_2-\mu_1=(3,3)^\top$, e passa pelo
ponto m\'edio $(\mu_1+\mu_2)/2=(1{,}5,1{,}5)^\top$. \'E a mediatriz dos
centros, como esperado para covari\^ancias iguais e isotr\'opicas.

\section*{Quest\~ao 3 --- Probabilidade de erro te\'orica e emp\'irica}

\paragraph{Enunciado.} Calcular a probabilidade de erro te\'orica
(atrav\'es das densidades a posteriori) e estim\'a-la pelo conjunto de
dados gerado na Quest\~ao~2.

\paragraph{Erro te\'orico.} A regra linear $h(\mathbf{x})=w^\top\mathbf{x}+b$,
com $w=\mu_2-\mu_1$ e $b=-\tfrac12(\mu_2^\top\mu_2-\mu_1^\top\mu_1)$,
projeta o problema em uma dimens\~ao. Sob $C_i$, $h$ \'e Gaussiana com
m\'edia $w^\top\mu_i+b$ e vari\^ancia $w^\top\Sigma w$. Definindo a
dist\^ancia de Mahalanobis
$\delta=\sqrt{(\mu_2-\mu_1)^\top\Sigma^{-1}(\mu_2-\mu_1)}$, mostra-se
que para priors iguais
\[
P_e^{\mathrm{Bayes}}=\Phi\!\bigl(-\tfrac{\delta}{2}\bigr).
\]
Para a configura\c{c}\~ao original, $\delta=\sqrt{18}$ e
\[
P_e^{\mathrm{Bayes}}=\Phi\!\bigl(-\sqrt{18}/2\bigr)=\Phi(-2{,}121)\approx 0{,}0169.
\]

\paragraph{Erro emp\'irico.} Aplicando $h(\mathbf{x})=3x_1+3x_2-9$ aos $2000$
pontos da Quest\~ao~2, obteve-se taxa de erro $0{,}0155$ ($31$
classifica\c{c}\~oes incorretas).

\paragraph{Coment\'ario.} O valor emp\'irico ($0{,}0155$) est\'a a menos
de meio desvio padr\~ao do te\'orico ($0{,}0169$); o desvio padr\~ao
amostral \'e $\sqrt{p(1-p)/n}\approx 0{,}003$ para $n=2000$. A
concord\^ancia confirma a deriva\c{c}\~ao da Quest\~ao~1.

\section*{Quest\~ao 5 --- Classificador Euclidiano}

\paragraph{Enunciado.} Projetar um classificador baseado na dist\^ancia
Euclidiana \`as m\'edias das classes (estimadas no conjunto de
treinamento). Achar a fun\c{c}\~ao discriminante te\'orica para esse
caso e a probabilidade de erro te\'orica; estimar a probabilidade de
erro pelo conjunto de teste.

\paragraph{Fun\c{c}\~ao discriminante te\'orica.} Decidir pela m\'edia
mais pr\'oxima equivale a maximizar
$-\tfrac12\|\mathbf{x}-\mu_i\|^2$. Expandindo e descartando o termo comum
$-\tfrac12\mathbf{x}^\top\mathbf{x}$,
\[
h_{\mathrm{Euc}}(\mathbf{x})=(\mu_2-\mu_1)^\top\mathbf{x}
-\tfrac12\bigl(\mu_2^\top\mu_2-\mu_1^\top\mu_1\bigr).
\]
Para $\mu_1=\mathbf{0}$ e $\mu_2=(3,3)^\top$, isto d\'a
$h_{\mathrm{Euc}}(\mathbf{x})=3x_1+3x_2-9$, \emph{coincidente com a regra
de Bayes}, pois $\Sigma=I$.

\paragraph{Erro te\'orico.} Como a regra te\'orica do Euclidiano
coincide com a de Bayes nesta configura\c{c}\~ao,
\[
P_e^{\text{Euc, te\'orico}}=\Phi(-\sqrt{18}/2)\approx 0{,}0169.
\]

\paragraph{Vers\~ao emp\'irica.} Substituindo $\mu_i$ pelas m\'edias
amostrais $\hat{\mu}_i$ obtidas no treino,
\[
\hat{h}_{\mathrm{Euc}}(\mathbf{x})\approx 2{,}9803\,x_1+2{,}9997\,x_2-9{,}0472.
\]
Avaliando-a no conjunto de teste, o erro foi $0{,}0125$.

\begin{figure}[ht]
\centering
\includegraphics[width=0.55\textwidth]{figures/configuracao_original_q2_q7/euclidiano.png}
\caption{Quest\~ao~5: classificador Euclidiano com m\'edias amostrais
sobre o conjunto de treino.}
\end{figure}

\paragraph{Coment\'ario.} Como $\Sigma=I$, a regra te\'orica do
Euclidiano \'e igual \`a de Bayes (mesma fronteira, mesmo erro
$0{,}0169$). A vers\~ao emp\'irica difere apenas pela
estima\c{c}\~ao das m\'edias com $1000$ pontos: o vetor normal
$(2{,}98,\,3{,}00)$ \'e quase igual a $(3,3)$, e o erro de teste
($0{,}0125$) coincide essencialmente com o de Bayes ($0{,}0120$).

\section*{Quest\~ao 6 --- Bayes ing\^enuo n\~ao param\'etrico}

\paragraph{Enunciado.} Projetar um classificador de Bayes ing\^enuo
estimando as densidades de probabilidade das classes de forma n\~ao
param\'etrica, sobre o conjunto de treinamento. Achar graficamente a
fun\c{c}\~ao discriminante e estimar o erro pelo conjunto de teste.

\paragraph{Modelo.} O classificador ing\^enuo assume independ\^encia
das componentes,
\[
p(\mathbf{x}\mid C_i)=\prod_{j=1}^{d}p_j(x_j\mid C_i),
\]
e cada marginal $p_j(\cdot\mid C_i)$ foi estimada por KDE Gaussiana
(\texttt{scipy.stats.gaussian\_kde}) com banda de Scott. A regra
usa o log-posteriori
$g_i(\mathbf{x})=\ln P(C_i)+\sum_j\ln p_j(x_j\mid C_i)$, e a fronteira
\'e o conjunto $\{\mathbf{x}:g_2(\mathbf{x})=g_1(\mathbf{x})\}$.

\begin{figure}[ht]
\centering
\includegraphics[width=0.55\textwidth]{figures/configuracao_original_q2_q7/bayes_ingenuo.png}
\caption{Quest\~ao~6: superf\'icie de separa\c{c}\~ao do Bayes ing\^enuo
n\~ao param\'etrico (KDE marginal).}
\end{figure}

\paragraph{Erro de teste.} $0{,}0155$.

\paragraph{Coment\'ario.} Aqui $\Sigma=I$ \'e diagonal, logo a
hip\'otese de independ\^encia das componentes \'e \emph{exata}; o
ing\^enuo recupera o classificador \'otimo a menos do ru\'\i do da
estima\c{c}\~ao KDE. O erro $0{,}0155$ fica entre o de Bayes
($0{,}0120$) e o do $k$-NN $k=5$, refletindo essa pequena perda.

\section*{Quest\~ao 8 --- Nova configura\c{c}\~ao das classes}

\paragraph{Enunciado.} Mudar as classes para
\[
C_1:\mathcal{N}\!\bigl(\mu_1=\tbinom{0}{0},\Sigma\bigr),\qquad
C_2:\mathcal{N}\!\bigl(\mu_2=\tbinom{2}{0},\Sigma\bigr),\qquad
\Sigma=\begin{bmatrix}1 & 0{,}6\\ 0{,}6 & 1\end{bmatrix},
\]
e refazer todas as quest\~oes anteriores.\\[4pt]
As mesmas sementes ($\mathtt{SEED\_Q2}=426$, $\mathtt{SEED\_Q4}=89$) e
o mesmo $n=1000$ s\~ao usados, alterando apenas $\mu_2$ e $\Sigma$. Cada
subse\c{c}\~ao a seguir refaz uma das quest\~oes anteriores.

% --- Q1 refeita ---
\subsection*{Refazendo a Quest\~ao 1: fun\c{c}\~ao discriminante}

A simplifica\c{c}\~ao da Quest\~ao~1 ainda vale para
$-\tfrac12\mathbf{x}^\top\Sigma^{-1}\mathbf{x}$ (termo comum, pois $\Sigma$ \'e o
mesmo nas duas classes), o que ainda leva a uma regra linear:
\[
h(\mathbf{x})=(\mu_2-\mu_1)^\top\Sigma^{-1}\mathbf{x}-
\tfrac12\bigl(\mu_2^\top\Sigma^{-1}\mu_2-\mu_1^\top\Sigma^{-1}\mu_1\bigr).
\]
Calculando $\Sigma^{-1}$ com $\det\Sigma=1-0{,}36=0{,}64$,
\[
\Sigma^{-1}=\frac{1}{0{,}64}\begin{bmatrix}1 & -0{,}6\\ -0{,}6 & 1\end{bmatrix}
=\begin{bmatrix}1{,}5625 & -0{,}9375\\ -0{,}9375 & 1{,}5625\end{bmatrix}.
\]
Como $\mu_1=\mathbf{0}$, $\Sigma^{-1}(\mu_2-\mu_1)=(3{,}125,-1{,}875)^\top$
e $\tfrac12\mu_2^\top\Sigma^{-1}\mu_2=\tfrac12(2)(3{,}125)=3{,}125$, de
onde
\[
h(\mathbf{x})=3{,}125\,x_1-1{,}875\,x_2-3{,}125,
\qquad
\begin{cases}
h(\mathbf{x})<0\Rightarrow C_1,\\
h(\mathbf{x})\ge 0\Rightarrow C_2.
\end{cases}
\]
A fronteira ainda \'e uma reta, por\'em \'e ortogonal a
$\Sigma^{-1}(\mu_2-\mu_1)$ \emph{na geometria de Mahalanobis} --- e n\~ao
mais ao vetor $\mu_2-\mu_1$ no plano Euclidiano.

% --- Q2 refeita ---

Com $\mathtt{SEED\_Q2}=426$ e a nova distribui\c{c}\~ao, geraram-se
$1000$ amostras por classe. As m\'edias amostrais s\~ao
$\hat{\mu}_1\approx(-0{,}0156,\,0{,}0169)^\top$ e
$\hat{\mu}_2\approx(2{,}0021,\,0{,}0343)^\top$.

\paragraph{Coment\'ario.} As isodensidades agora s\~ao elipses
inclinadas (correla\c{c}\~ao $0{,}6$) e os centros distam apenas

tem componente $x_2$ negativa: a correla\c{c}\~ao positiva entre as
vari\'aveis ``inclina'' a fronteira na dire\c{c}\~ao oposta \`a do
gradiente de uma simples diferen\c{c}a de m\'edias.

% --- Q3 refeita ---
\subsection*{Refazendo a Quest\~ao 3: erro de Bayes}

Com a mesma f\'ormula geral
$P_e^{\mathrm{Bayes}}=\Phi(-\delta/2)$ e
\[
\delta^2=(\mu_2-\mu_1)^\top\Sigma^{-1}(\mu_2-\mu_1)
=(2,0)\binom{3{,}125}{-1{,}875}=6{,}25,\qquad \delta=2{,}5,
\]
\[
P_e^{\mathrm{Bayes}}=\Phi(-1{,}25)\approx 0{,}1056.
\]
O erro emp\'irico nos dados da Quest\~ao~2 desta parte foi $0{,}1060$.

\paragraph{Coment\'ario.} O erro \'e $\sim 6$ vezes maior do que na
configura\c{c}\~ao original ($0{,}1056$ vs.\ $0{,}0169$): as classes
ficaram mais pr\'oximas e a correla\c{c}\~ao reduz a separabilidade
\emph{efetiva} (Mahalanobis $2{,}5$ vs.\ Euclidiana $\sqrt{18}\approx 4{,}24$
do problema original).

% --- Q5 refeita ---
\subsection*{Refazendo a Quest\~ao 5: classificador Euclidiano}

\paragraph{Fun\c{c}\~ao te\'orica.} A regra Euclidiana com m\'edias
verdadeiras \emph{ignora} $\Sigma$:
\[
h_{\mathrm{Euc}}(\mathbf{x})=(\mu_2-\mu_1)^\top\mathbf{x}-\tfrac12(\mu_2^\top\mu_2-\mu_1^\top\mu_1)
=2x_1-2.
\]
A fronteira \'e a reta vertical $x_1=1$.

\paragraph{Erro te\'orico.} Sob $C_i$, $h$ \'e Gaussiana com m\'edia
$w^\top\mu_i+b$ e vari\^ancia $w^\top\Sigma w$. Aqui $w=(2,0)^\top$ e
$b=-2$, portanto $w^\top\Sigma w=4$, $s=2$, $m_1=-2$ e $m_2=2$. Logo
\[
P_e^{\text{Euc, te\'orico}}
=\tfrac12\,\bigl[1-\Phi(1)\bigr]+\tfrac12\,\Phi(-1)
=\Phi(-1)\approx 0{,}1587.
\]
\emph{O Euclidiano deixa de ser \'otimo}: o gap $0{,}1587-0{,}1056=0{,}0531$
\'e o pre\c{c}o de ignorar a correla\c{c}\~ao.

\paragraph{Vers\~ao emp\'irica.} Com as m\'edias amostrais,
$\hat{h}_{\mathrm{Euc}}(\mathbf{x})\approx 2{,}0177\,x_1+0{,}0174\,x_2-2{,}0045$.
Erro no teste: $0{,}1645$.

\begin{figure}[ht]
\centering
\includegraphics[width=0.55\textwidth]{figures/questao_8__nova_configuracao/euclidiano.png}
\caption{Q8/Q5: classificador Euclidiano emp\'irico (m\'edias amostrais)
sobre o conjunto de treino da nova configura\c{c}\~ao.}
\end{figure}

\paragraph{Coment\'ario.} A vers\~ao emp\'irica fica ligeiramente pior
que a te\'orica (erro $0{,}1645$ vs.\ $0{,}1587$) por causa da
estima\c{c}\~ao das m\'edias. O fato qualitativo \'e que ambas
\emph{deixam} cerca de $5$ pontos percentuais sobre a mesa em rela\c{c}\~ao
ao \'otimo de Bayes ($0{,}1056$).

\subsection*{Refazendo a Quest\~ao 6: Bayes ing\^enuo n\~ao param\'etrico}

Mesmo modelo da Quest\~ao~6: KDE Gaussiana marginal por classe e
log-posterior produto. Erro de teste: $0{,}1620$.

\section*{S\'\i ntese final}

Comparando as duas configura\c{c}\~oes:
\begin{itemize}
\item Com $\Sigma=I$ e centros bem separados (Parte~I), todos os
classificadores ficam pr\'oximos do \'otimo de Bayes ($\sim 1{,}5\%$),
inclusive o Euclidiano e o ing\^enuo, porque ambas as suas hip\'oteses
de simplifica\c{c}\~ao (covari\^ancia identidade e independ\^encia
marginal) s\~ao satisfeitas.
\item Com $\Sigma$ correlacionado e centros mais pr\'oximos
(Parte~II), Bayes (que usa $\Sigma^{-1}$) atinge $\sim 10{,}6\%$
te\'orico, enquanto Euclidiano e ing\^enuo, que ignoram a
correla\c{c}\~ao, ficam acima de $15\%$. \'E o pre\c{c}o de modelos com
hip\'oteses incorretas.
\item O $k$-NN, n\~ao param\'etrico, recupera a estrutura local em
ambas as configura\c{c}\~oes; com $k=11$ atinge desempenho
compar\'avel ao de Bayes nos dois cen\'arios.
\end{itemize}
